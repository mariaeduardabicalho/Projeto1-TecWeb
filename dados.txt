 CREATE TABLE NotasTESTE2(id  INT NOT NULL AUTO_INCREMENT,
tipo_doc  VARCHAR(5) NOT NULL,
categoria VARCHAR(50) ,
conteudo VARCHAR(1000) CHARACTER SET utf8, data_postagem  DATE,
 nome_doc VARCHAR(50) NOT NULL,
 PRIMARY KEY(id)
 );



CREATE TABLE usuario (
 id INT  NOT NULL  AUTO_INCREMENT, primeiro_nome VARCHAR(50) NOT NULL, sobrenome VARCHAR(50) NOT NULL ,
 email VARCHAR(70) NOT NULL ,
 tipo VARCHAR(9),
 usuario VARCHAR(15) NOT NULL, 
senha VARCHAR(50) NOT NULL, 
PRIMARY KEY (ID) 
 );
